 73 -def click( x = None, y = None, button = "bc1" ): 

 74      """ 
 75      Execute Mouse Clicks. If the mouse is dragging an object 
 76      then the release click will be performed. 
 77   
 78      Arguments: 
 79      - x: The X coordinate in the screen. 
 80      - y: The Y coordinate in the screen. 
 81      - button: The button click that has to be performed. 
 82   
 83      Return True 
 84      """ 
 85   
 86      global isGnome 
 87      global dragging 
 88   
 89      if not x or not y: 
 90          x, y = position() 
 91   
 92      if dragging: 
 93          button = button[:2] + 'r' 
 94          dragging = False 
 95      elif button[2] == 'p': 
 96          dragging = True 
 97   
 98      if isGnome: 
 99          try: 
100              reg.generateMouseEvent( x, y, button ) 
101          except: 
102              isGnome = False 
103      else: 
104          for action in clickType[button[2]]: 
105              xDisplay.xtest_fake_input(action, int(button[1]), clickVal[action]) 
106          xDisplay.flush() 
107   
108      return True
